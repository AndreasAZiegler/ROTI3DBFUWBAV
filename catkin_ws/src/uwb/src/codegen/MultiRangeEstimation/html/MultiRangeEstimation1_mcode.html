1%% Compute time-of-flights, ranges, clock offsets and skews for multi-unit UWB tracker
  2function [tofs, ranges, clock_offsets, clock_skews] = MultiRangeEstimation( ...
  3    timestamp_master_request_1_recv, ...
  4    timestamp_slave_reply_send, ...
  5    timestamp_master_request_2_recv, ...
  6    timestamp_master_request_1_array, ...
  7    timestamp_slave_reply_array, ...
  8    timestamp_master_request_2_array, ...
  9    params ...
 10)
 11
 12    coder.cstructname(params, 'MultiRangeEstimationParams');
 13
 14    assert(numel(timestamp_master_request_1_array) == numel(timestamp_master_request_2_array));
 15    assert(numel(timestamp_master_request_1_array) == numel(timestamp_slave_reply_array));
 16
 17    num_of_units = numel(timestamp_master_request_1_array);
 18
 19%     if ~coder.target('MATLAB')
 20%         coder.ceval('printf', ['num_of_units: %f\n' 10 0], num_of_units);
 21%         coder.ceval('printf', ['timestamp_master_request_1_recv: %f\n' 10 0], timestamp_master_request_1_recv);
 22%         coder.ceval('printf', ['timestamp_master_request_1_array(1): %f\n' 10 0], timestamp_master_request_1_array(1));
 23%     end
 24
 25    tofs = zeros(1, num_of_units);
 26    ranges = zeros(1, num_of_units);
 27    clock_offsets = zeros(1, num_of_units + 1);
 28    clock_skews = zeros(1, num_of_units + 1);
 29
 30    [primary_tof, primary_range] = ProcessPrimaryMeasurement( ...
 31        timestamp_master_request_1_recv, ...
 32        timestamp_slave_reply_send, ...
 33        timestamp_master_request_2_recv, ...
 34        timestamp_master_request_1_array, ...
 35        timestamp_slave_reply_array, ...
 36        timestamp_master_request_2_array, ...
 37        params ...
 38    );
 39
 40%     if ~coder.target('MATLAB')
 41%         coder.ceval('printf', ['primary_tof: %f' 10 0], primary_tof);
 42%     end
 43
 44    tofs(1) = primary_tof;
 45    ranges(1) = primary_range;
 46
 47    [slave_clock_offset, slave_clock_skew] = ProcessSlaveMeasurement( ...
 48        primary_range, ...
 49        timestamp_master_request_1_recv, ...
 50        timestamp_master_request_2_recv, ...
 51        timestamp_master_request_1_array, ...
 52        timestamp_master_request_2_array, ...
 53        params ...
 54    );
 55    clock_offsets(end) = slave_clock_offset;
 56    clock_skews(end) = slave_clock_skew;
 57
 58    % Compute slave processing time
 59    adjusted_processing_time_slave_us = ConvertDWTimeunitsToMicroseconds( ...
 60        timestamp_slave_reply_array(1) - timestamp_master_request_1_array(1), params) ...
 61        - 2 * primary_tof;
 62    % Alternative: Use slave measurements and clock_skew to compute slave
 63    % processing time
 64%     adjusted_processing_time_slave = double( ...
 65%         timestamp_slave_reply_send - timestamp_master_request_1_recv ...
 66%     ) * (1 + slave_clock_skew);
 67
 68    for index = 2:num_of_units
 69        [clock_offset, clock_skew] = ProcessSecondaryMeasurement( ...
 70            index, ...
 71            timestamp_master_request_1_array, ...
 72            timestamp_master_request_2_array, ...
 73            params ...
 74        );
 75
 76        timediff_primary_slave_secondary_us = ConvertDWTimeunitsToMicroseconds( ...
 77            timestamp_slave_reply_array(index) - clock_offset - timestamp_master_request_1_array(1), params ...
 78        ) * (1 + clock_skew);
 79
 80        tof_secondary = ...
 81            timediff_primary_slave_secondary_us - primary_tof - adjusted_processing_time_slave_us;
 82        range_secondary = ConvertTimeOfFlightToDistance(tof_secondary, params);
 83
 84        tofs(index) = tof_secondary;
 85        ranges(index) = range_secondary;
 86        clock_offsets(index) = clock_offset;
 87        clock_skews(index) = clock_skew;
 88    end
 89
 90end
